case 'ban': {
  await interaction.deferReply();

  const user = options.getUser('usuario');
  const reason = options.getString('motivo') || 'Não informado';

  if (!interaction.member?.permissions?.has(PermissionsBitField.Flags.BanMembers)) {
    return interaction.editReply({ content: '❌ Você não tem permissão para banir membros.' });
  }

  const member = interaction.guild.members.cache.get(user.id);
  if (!member) {
    return interaction.editReply({ content: '❌ Membro não encontrado no servidor.' });
  }

  try {
    await member.ban({ reason });

    await interaction.editReply(`🔨 ${member.user.tag} foi banido.`);

    // 📥 ENVIA LOG DE BAN NO CANAL
    const logChannel = interaction.guild.channels.cache.get('1401705638668669149');
    if (logChannel && logChannel.isTextBased()) {
      const embed = new EmbedBuilder()
        .setTitle('🔨 | Membro Banido')
        .addFields(
          { name: '👤 Membro', value: `${user.tag} (\`${user.id}\`)`, inline: true },
          { name: '🛡️ Moderador', value: `${interaction.user.tag}`, inline: true },
          { name: '📄 Motivo', value: reason, inline: false }
        )
        .setColor(0xff0000)
        .setImage('https://media.tenor.com/kCbfYkT1BvQAAAAC/ban-hammer.gif')
        .setTimestamp();

      await logChannel.send({ embeds: [embed] });
    }

  } catch (error) {
    console.error('Erro ao banir o membro:', error);
    await interaction.editReply('❌ Ocorreu um erro ao banir esse membro.');
  }

  break;
}

case 'mute': {
  await interaction.deferReply();

  const member = options.getMember('usuario');
  const duration = options.getString('tempo');
  const reason = options.getString('motivo') || 'Não informado';

  if (!interaction.member.permissions.has(PermissionsBitField.Flags.ModerateMembers)) {
    return interaction.editReply({ content: '🚫 Você não tem permissão para mutar membros.' });
  }

  if (!member) {
    return interaction.editReply({ content: '🚫 Usuário não encontrado.' });
  }

  if (!member.moderatable) {
    return interaction.editReply({ content: '🚫 Não consigo mutar este usuário.' });
  }

  if (!duration) {
    return interaction.editReply({ content: '⏰ Informe o tempo (ex: 10s, 5m, 1h).' });
  }

  const time = ms(duration);

  if (!time || time < 1000) {
    return interaction.editReply({ content: '❌ Formato inválido. Use: 10s, 5m, 1h, 1d.' });
  }

  try {
    await member.timeout(time, reason);

    const embedResponse = new EmbedBuilder()
      .setTitle('✅ | Membro Mutado com Sucesso')
      .setDescription(`O membro ${member} foi mutado por **${duration}**.`)
      .addFields(
        { name: '📄 Motivo', value: reason },
        { name: '🛡️ Moderador', value: interaction.user.tag }
      )
      .setColor(0x00b0f4)
      .setTimestamp();

    await interaction.editReply({ embeds: [embedResponse] });

    // 📥 ENVIA LOG NO CANAL DE LOGS
    const logChannel = interaction.guild.channels.cache.get('1402083773264298074');
    if (logChannel && logChannel.isTextBased()) {
      const embedLog = new EmbedBuilder()
        .setTitle('🔇 | Membro Mutado')

case 'mutecall': {
  await interaction.deferReply();

  const member = options.getMember('usuario');
  const tempo = options.getString('tempo');
  const motivo = options.getString('motivo') || 'Não informado';

  if (!interaction.member?.permissions?.has(PermissionsBitField.Flags.MuteMembers)) {
    return interaction.editReply({ content: '🚫 Você não tem permissão para mutar membros na call.' });
  }

  if (!member?.voice?.channel) {
    return interaction.editReply({ content: '🚫 O usuário não está em um canal de voz.' });
  }

  const durationMs = tempo ? ms(tempo) : null;

  try {
    await member.voice.setMute(true);
    
    const replyEmbed = new EmbedBuilder()
      .setTitle('🔇 | Membro Mutado na Call')
      .addFields(
        { name: '👤 Membro', value: `${member.user.tag} (\`${member.id}\`)`, inline: true },
        { name: '🛡️ Moderador', value: `${interaction.user.tag}`, inline: true },
        { name: '📄 Motivo', value: motivo, inline: false },
        ...(durationMs ? [{ name: '⏰ Tempo', value: tempo, inline: true }] : [])
      )
      .setColor(0xffa500)
      .setImage('https://media.tenor.com/2kABvuoN3y0AAAAC/shhh-silence.gif')
      .setTimestamp();

    await interaction.editReply({ embeds: [replyEmbed] });

    // ✅ ENVIA LOG PARA O CANAL
    const logChannel = interaction.guild.channels.cache.get('1402083773264298074');
    if (logChannel?.isTextBased()) {
      await logChannel.send({ embeds: [replyEmbed] });
    }

    // 🔁 DESMUTE AUTOMÁTICO
    if (durationMs) {
      setTimeout(async () => {
        if (member.voice.channel) {
          await member.voice.setMute(false);
          const unmuteEmbed = new EmbedBuilder()
            .setTitle('🔊 | Desmute Automático')
            .setDescription(`O usuário ${member.user.tag} foi desmutado automaticamente após \`${tempo}\`.`)
            .setColor(0x00ff00)
            .setTimestamp();

          await interaction.followUp({ embeds: [unmuteEmbed] });

          if (logChannel?.isTextBased()) {
            await logChannel.send({ embeds: [unmuteEmbed] });
          }
        }
      }, durationMs);
    }

  } catch (err) {
    console.error('Erro ao mutar na call:', err);
    await interaction.editReply({ content: '❌ Erro ao tentar mutar o usuário na call.' });
  }

  break;
}
